$userlist=get-content C:\edr\pwd_never_expires.txt

foreach ($i in $userlist)
{
    Set-Aduser -Identity $i -ChangePasswordAtLogon $true
    $a=Get-ADUser -Identity $i -Properties * | Select-Object pwdlastset
    if ($a.pwdlastset -eq 0)
    {
       Add-Content -Path "C:\edr\pwd_nvr_exp_log3.txt" -Value "$i  done"
    }
    else
    {
         Add-Content -Path "C:\edr\pwd_nvr_exp_log4.txt" -Value "$i not ok"
    }
}
